[docs]def header(x, n=70, filler="=", pop=False):
    """print the content `x`, align centered, with filler `n` of length `n` """
    if x is None or len(x) == 0:
        f = filler*n
    else:
        x     = str(x)
        l     = (n-len(x)-2)/2
        front = int(l)
        end   = int(l) if l%1 == 0 else int(l)+1
        f     = filler*front+" "+x+" "+filler*end
    return printc(f,pop=pop)



[docs]def printc(x, color=None, pop=None):
    """
    if pop is True:
        return only the string `x` or `colored(x)`
    else:
        display `x` or `colored(x)`"""
    color = color or '{}'
    pop   = pop or VISUAL_CONFIG.return_string
    if x is None:
        return
    x = color.format(x) if VISUAL_CONFIG.add_printout_color else x
    if pop:
        return x
    print(x)




[docs]def join_str(*args):
    if len(args) == 0 or args[0] is None:
        return
    return ' '.join([str(i) for i in args])
